Alphabet

# import libraries
import gtts, string, IPython, os  
import ipywidgets as widgets

os.makedirs("alphabet_audio", exist_ok = True) # make folder for alphabet audio
letters = list(string.ascii_lowercase) # get letters (ascii is ok because alphabets are the same)

def save_text_mp3(text: str, loc: str, lang: str = "it"): gtts.gTTS(text, lang = "it").save(loc)    
for l in letters: save_text_mp3(l, f"alphabet_audio/{l}.mp3") # save all letters
for l in ["a", "c", "j", "k", "h", "i"]:
    display(l, IPython.display.Audio(f"alphabet_audio/{l}.mp3"))
